public class ArrayOverflowExample {
    public static void main(String[] args) {
        int[] array = new int[5]; // Create an array of size 5

        try {
            // Access or modify elements beyond the array size
            for (int i = 0; i <= array.length; i++) {
                array[i] = i;
            }
        } catch (ArrayIndexOutOfBoundsException e) {
            System.out.println("ArrayOverflowException occurred!");
            System.out.println("Please ensure that you do not exceed the array size.");
        }
    }
}
